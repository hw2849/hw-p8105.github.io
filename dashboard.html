## scatterplot 
my_data %>%     
  filter( ## detect all korean restaurant
    str_detect(cuisine_description, "[Kk][Oo][Rr][Ee][Aa][Nn]")
    ) %>% 
  mutate( ## the tiny label showing grade and dba
    text_label = str_c("Type:", dba, "\nGrade: ", grade) 
  ) %>% 
  plot_ly(
    x = ~inspection_date, 
    y = ~score, 
    type = "scatter", 
    mode = "markers", 
    color = ~district, 
    text = ~text_label, 
    alpha = .3
  )
